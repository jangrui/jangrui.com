## 1.创建nginx的configMap
```shell
[root@master ~]# kubectl create configmap confnginx --from-file nginx.conf
configmap/confnginx created
# 查看是否创建成功
[root@master ~]# kubectl get configmap 
NAME         DATA   AGE
confnginx    1      23s
redis-conf   1      23h
[root@master ~]# 
```
